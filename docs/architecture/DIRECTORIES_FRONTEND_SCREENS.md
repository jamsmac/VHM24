## Overview

The Directory System frontend consists of management screens for administrators and embedded components (like inline create) for all users across the application.

### Design Principles

1. **Wizard-based creation** - Step-by-step directory creation
2. **Live preview** - Show how entries will look while configuring
3. **Inline create everywhere** - Add missing values without context switch
4. **Clear origin indication** - Always show OFFICIAL vs LOCAL badges
5. **Mobile-responsive** - All screens work on tablet/mobile

---

## Route Structure

```
/dashboard/directories                    # Directory list
/dashboard/directories/new                # Create new directory (wizard)
/dashboard/directories/templates          # Template gallery
/dashboard/directories/:code              # Directory detail + entries
/dashboard/directories/:code/settings     # Directory settings
/dashboard/directories/:code/sources      # Data sources management
/dashboard/directories/:code/entries/:id  # Entry card (detail view)
/dashboard/directories/:code/entries/new  # Create new entry
```

// src/store/api/directoriesApi.ts

import { createApi, fetchBaseQuery } from '@reduxjs/toolkit/query/react';

export const directoriesApi = createApi({
  reducerPath: 'directoriesApi',
  baseQuery: fetchBaseQuery({ baseUrl: '/api' }),
  tagTypes: ['Directory', 'DirectoryEntry', 'DirectorySource'],
  endpoints: (builder) => ({
    // Directories
    getDirectories: builder.query<Directory[], DirectoryFilters>({
      query: (filters) => ({
        url: '/directories',
        params: filters,
      }),
      providesTags: ['Directory'],
    }),

    getDirectory: builder.query<Directory, string>({
      query: (code) => `/directories/code/${code}`,
      providesTags: (result, error, code) => [{ type: 'Directory', id: code }],
    }),

    createDirectory: builder.mutation<Directory, CreateDirectoryDto>({
      query: (body) => ({
        url: '/directories',
        method: 'POST',
        body,
      }),
      invalidatesTags: ['Directory'],
    }),

    // Entries
    getDirectoryEntries: builder.query<
      PaginatedResponse<DirectoryEntry>,
      { directoryCode: string; params?: EntryFilters }
    >({
      query: ({ directoryCode, params }) => ({
        url: `/directories/${directoryCode}/entries`,
        params,
      }),
      providesTags: (result, error, { directoryCode }) => [
        { type: 'DirectoryEntry', id: directoryCode },
      ],
    }),

    createDirectoryEntry: builder.mutation<
      DirectoryEntry,
      { directoryCode: string; body: CreateEntryDto }
    >({
      query: ({ directoryCode, body }) => ({
        url: `/directories/${directoryCode}/entries`,
        method: 'POST',
        body,
      }),
      invalidatesTags: (result, error, { directoryCode }) => [
        { type: 'DirectoryEntry', id: directoryCode },
      ],
    }),

    // Quick create for inline
    quickCreateEntry: builder.mutation<
      DirectoryEntry,
      { directoryCode: string; name: string }
    >({
      query: ({ directoryCode, name }) => ({
        url: `/directories/${directoryCode}/entries/quick`,
        method: 'POST',
        body: { name_ru: name },
      }),
      invalidatesTags: (result, error, { directoryCode }) => [
        { type: 'DirectoryEntry', id: directoryCode },
      ],
    }),

    // Sources
    syncSource: builder.mutation<
      SyncLog,
      { directoryId: string; sourceId: string }
    >({
      query: ({ directoryId, sourceId }) => ({
        url: `/directories/${directoryId}/sources/${sourceId}/sync`,
        method: 'POST',
      }),
      invalidatesTags: (result, error, { directoryId }) => [
        { type: 'DirectoryEntry', id: directoryId },
        { type: 'DirectorySource', id: directoryId },
      ],
    }),
  }),
});

export const {
  useGetDirectoriesQuery,
  useGetDirectoryQuery,
  useCreateDirectoryMutation,
  useGetDirectoryEntriesQuery,
  useCreateDirectoryEntryMutation,
  useQuickCreateEntryMutation,
  useSyncSourceMutation,
} = directoriesApi;

    const result = await quickCreate({ directoryCode, name }).unwrap();
    return result;
  };

  return {
    entries: entries?.data || [],
    isLoading,
    search,
    setSearch,
    handleCreate,
    isCreating,
  };
}
